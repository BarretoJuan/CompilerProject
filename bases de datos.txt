

CREATE TABLE oficinas (idOficinas int(4) primary key auto_increment, ciudad varchar(45), estado varchar(20), metas decimal(5,2), telefono varchar(11) );

CREATE TABLE vendedores (idVendedores int(4) PRIMARY KEY AUTO_INCREMENT, nombre varchar(45), idOficinas int(4), cuota decimal(5,2), ventas decimal(5,2), FOREIGN KEY (idOficinas) REFERENCES oficinas(idOficinas));

CREATE TABLE clientes (idClientes int(4) PRIMARY KEY AUTO_INCREMENT, nombre varchar(45), idVendedores int(4), limite decimal(5,2) , FOREIGN KEY (idVendedores) references vendedores(idvendedores));

CREATE TABLE ordenes (idOrdenes int(4) PRIMARY KEY AUTO_INCREMENT, Cliente int(4) UNIQUE NOT NULL, producto varchar(45) not null, cantidad int(4) not null, monto decimal(5,2) not null, descuento decimal(3,2) not null, FOREIGN KEY(Cliente) references clientes(idClientes));
